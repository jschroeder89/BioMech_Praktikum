666 #define MEM_ALIGN_MASK(a) ((size_t)(a) - 1U)
667 
668 /**
669  * @brief Aligns to the previous aligned memory address.
670  *
671  * @param[in] p variable to be aligned
672  * @param[in] a alignment, must be a power of two
673  */
674 #define MEM_ALIGN_PREV(p, a) ((size_t)(p) & ~MEM_ALIGN_MASK(a))
675 
676 /**
677  * @brief Aligns to the new aligned memory address.
678  *
679  * @param[in] p variable to be aligned
680  * @param[in] a alignment, must be a power of two
681  */
682 #define MEM_ALIGN_NEXT(p, a) MEM_ALIGN_PREV((size_t)(p) + \
683  MEM_ALIGN_MASK(a), (a))
684 
685 /**
686  * @brief Returns whatever a pointer or memory size is aligned.
687  *
688  * @param[in] p variable to be aligned
689  * @param[in] a alignment, must be a power of two
690  */
691 #define MEM_IS_ALIGNED(p, a) (((size_t)(p) & MEM_ALIGN_MASK(a)) == 0U)
692 
693 /**
694  * @brief Returns whatever a constant is a valid alignment.
695  * @details Valid alignments are powers of two.
696  *
697  * @param[in] a alignment to be checked, must be a constant
698  */
699 #define MEM_IS_VALID_ALIGNMENT(a) \
700  (((size_t)(a) != 0U) && (((size_t)(a) & ((size_t)(a) - 1U)) == 0U))
