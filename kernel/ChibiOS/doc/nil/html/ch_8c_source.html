209 /**
210  * @brief Initializes the kernel.
211  * @details Initializes the kernel structures, the current instructions flow
212  * becomes the idle thread upon return. The idle thread must not
213  * invoke any kernel primitive able to change state to not runnable.
214  * @note This function assumes that the @p nil global variable has been
215  * zeroed by the runtime environment. If this is not the case then
216  * make sure to clear it before calling this function.
217  *
218  * @special
219  */
220 void chSysInit(void) {
221  thread_t *tp;
222  const thread_config_t *tcp;
223 
224 #if CH_DBG_SYSTEM_STATE_CHECK == TRUE
225  nil.isr_cnt = (cnt_t)0;
226  nil.lock_cnt = (cnt_t)0;
227 #endif
228 
229  /* System initialization hook.*/
230  CH_CFG_SYSTEM_INIT_HOOK();
231 
232  /* Iterates through the list of defined threads.*/
233  tp = &nil.threads[0];
234  tcp = nil_thd_configs;
235  while (tp < &nil.threads[CH_CFG_NUM_THREADS]) {
236 #if CH_DBG_ENABLE_STACK_CHECK
237  tp->wabase = (stkalign_t *)tcp->wbase;
238 #endif
239 
240  /* Port dependent thread initialization.*/
241  PORT_SETUP_CONTEXT(tp, tcp->wbase, tcp->wend, tcp->funcp, tcp->arg);
242 
243  /* Initialization hook.*/
244  CH_CFG_THREAD_EXT_INIT_HOOK(tp);
245 
246  tp++;
247  tcp++;
248  }
249 
250 #if CH_DBG_ENABLE_STACK_CHECK
251  /* The idle thread is a special case because its stack is set up by the
252  runtime environment.*/
253  tp->wabase = THD_IDLE_BASE;
254 #endif
255 
256  /* Interrupts partially enabled. It is equivalent to entering the
257  kernel critical zone.*/
258  chSysSuspend();
259 #if CH_DBG_SYSTEM_STATE_CHECK == TRUE
260  nil.lock_cnt = (cnt_t)1;
261 #endif
262 
263  /* Memory core initialization, if enabled.*/
264 #if CH_CFG_USE_MEMCORE == TRUE
265  _core_init();
266 #endif
267 
268  /* Heap initialization, if enabled.*/
269 #if CH_CFG_USE_HEAP == TRUE
270  _heap_init();
271 #endif
272 
273  /* Port layer initialization last because it depend on some of the
274  initializations performed before.*/
275  port_init();
276 
277  /* Runs the highest priority thread, the current one becomes the idle
278  thread.*/
279  nil.current = nil.next = nil.threads;
280  port_switch(nil.current, tp);
281  chSysUnlock();
282 }

314 /**
315  * @brief Time management handler.
316  * @note This handler has to be invoked by a periodic ISR in order to
317  * reschedule the waiting threads.
318  *
319  * @iclass
320  */
321 void chSysTimerHandlerI(void) {
322 
323  chDbgCheckClassI();
324 
325 #if CH_CFG_ST_TIMEDELTA == 0
326  thread_t *tp = &nil.threads[0];
327  nil.systime++;
328  do {
329  /* Is the thread in a wait state with timeout?.*/
330  if (tp->timeout > (systime_t)0) {
331 
332  chDbgAssert(!NIL_THD_IS_READY(tp), "is ready");
333 
334  /* Did the timer reach zero?*/
335  if (--tp->timeout == (systime_t)0) {
336  /* Timeout on queues/semaphores requires a special handling because
337  the counter must be incremented.*/
338  /*lint -save -e9013 [15.7] There is no else because it is not needed.*/
339  if (NIL_THD_IS_WTQUEUE(tp)) {
340  tp->u1.semp->cnt++;
341  }
342  else if (NIL_THD_IS_SUSP(tp)) {
343  *tp->u1.trp = NULL;
344  }
345  /*lint -restore*/
346  (void) chSchReadyI(tp, MSG_TIMEOUT);
347  }
348  }
349  /* Lock released in order to give a preemption chance on those
350  architectures supporting IRQ preemption.*/
351  chSysUnlockFromISR();
352  tp++;
353  chSysLockFromISR();
354  } while (tp < &nil.threads[CH_CFG_NUM_THREADS]);
355 #else
356  thread_t *tp = &nil.threads[0];
357  systime_t next = (systime_t)0;
358 
359  chDbgAssert(nil.nexttime == port_timer_get_alarm(), "time mismatch");
360 
361  do {
362  systime_t timeout = tp->timeout;
363 
364  /* Is the thread in a wait state with timeout?.*/
365  if (timeout > (systime_t)0) {
366 
367  chDbgAssert(!NIL_THD_IS_READY(tp), "is ready");
368  chDbgAssert(timeout >= (nil.nexttime - nil.lasttime), "skipped one");
369 
370  /* The volatile field is updated once, here.*/
371  timeout -= nil.nexttime - nil.lasttime;
372  tp->timeout = timeout;
373 
374  if (timeout == (systime_t)0) {
375  /* Timeout on thread queues requires a special handling because the
376  counter must be incremented.*/
377  if (NIL_THD_IS_WTQUEUE(tp)) {
378  tp->u1.tqp->cnt++;
379  }
380  else {
381  if (NIL_THD_IS_SUSP(tp)) {
382  *tp->u1.trp = NULL;
383  }
384  }
385  (void) chSchReadyI(tp, MSG_TIMEOUT);
386  }
387  else {
388  if (timeout <= (systime_t)(next - (systime_t)1)) {
389  next = timeout;
390  }
391  }
392  }
393 
394  /* Lock released in order to give a preemption chance on those
395  architectures supporting IRQ preemption.*/
396  chSysUnlockFromISR();
397  tp++;
398  chSysLockFromISR();
399  } while (tp < &nil.threads[CH_CFG_NUM_THREADS]);
400 
401  nil.lasttime = nil.nexttime;
402  if (next > (systime_t)0) {
403  nil.nexttime += next;
404  port_timer_set_alarm(nil.nexttime);
405  }
406  else {
407  /* No tick event needed.*/
408  port_timer_stop_alarm();
409  }
410 #endif
411 }

607 /**
608  * @brief Puts the current thread to sleep into the specified state with
609  * timeout specification.
610  * @details The thread goes into a sleeping state, if it is not awakened
611  * explicitly within the specified system time then it is forcibly
612  * awakened with a @p NIL_MSG_TMO low level message.
613  *
614  * @param[in] newstate the new thread state or a semaphore pointer
615  * @param[in] timeout the number of ticks before the operation timeouts.
616  * the following special values are allowed:
617  * - @a TIME_INFINITE no timeout.
618  * .
619  * @return The wakeup message.
620  * @retval NIL_MSG_TMO if a timeout occurred.
621  *
622  * @sclass
623  */
624 msg_t chSchGoSleepTimeoutS(tstate_t newstate, systime_t timeout) {
625  thread_t *ntp, *otp = nil.current;
626 
627  chDbgCheckClassS();
628 
629  chDbgAssert(otp != &nil.threads[CH_CFG_NUM_THREADS],
630  "idle cannot sleep");
631 
632  /* Storing the wait object for the current thread.*/
633  otp->state = newstate;
634 
635 #if CH_CFG_ST_TIMEDELTA > 0
636  if (timeout != TIME_INFINITE) {
637  systime_t abstime;
638 
639  /* TIMEDELTA makes sure to have enough time to reprogram the timer
640  before the free-running timer counter reaches the selected timeout.*/
641  if (timeout < (systime_t)CH_CFG_ST_TIMEDELTA) {
642  timeout = (systime_t)CH_CFG_ST_TIMEDELTA;
643  }
644 
645  /* Absolute time of the timeout event.*/
646  abstime = chVTGetSystemTimeX() + timeout;
647 
648  if (nil.lasttime == nil.nexttime) {
649  /* Special case, first thread asking for a timeout.*/
650  port_timer_start_alarm(abstime);
651  nil.nexttime = abstime;
652  }
653  else {
654  /* Special case, there are already other threads with a timeout
655  activated, evaluating the order.*/
656  if (chVTIsTimeWithinX(abstime, nil.lasttime, nil.nexttime)) {
657  port_timer_set_alarm(abstime);
658  nil.nexttime = abstime;
659  }
660  }
661 
662  /* Timeout settings.*/
663  otp->timeout = abstime - nil.lasttime;
664  }
665 #else
666 
667  /* Timeout settings.*/
668  otp->timeout = timeout;
669 #endif
670 
671  /* Scanning the whole threads array.*/
672  ntp = nil.threads;
673  while (true) {
674  /* Is this thread ready to execute?*/
675  if (NIL_THD_IS_READY(ntp)) {
676  nil.current = nil.next = ntp;
677  if (ntp == &nil.threads[CH_CFG_NUM_THREADS]) {
678  CH_CFG_IDLE_ENTER_HOOK();
679  }
680  port_switch(ntp, otp);
681  return nil.current->u1.msg;
682  }
683 
684  /* Points to the next thread in lowering priority order.*/
685  ntp++;
686  chDbgAssert(ntp <= &nil.threads[CH_CFG_NUM_THREADS],
687  "pointer out of range");
688  }
689 }
