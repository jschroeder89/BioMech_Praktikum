232  * @brief Seconds to system ticks.
233  * @details Converts from seconds to system ticks number.
234  * @note The result is rounded upward to the next tick boundary.
235  * @note This function uses a 64 bits internal representation,
236  * use with non-constant parameters can lead to inefficient
237  * code because 64 bits arithmetic would be used at runtime.
238  *
239  * @param[in] sec number of seconds
240  * @return The number of ticks.

242  * @api
243  */

491 /**
492  * @brief Returns the time interval until the next timer event.
493  * @note The return value is not perfectly accurate and can report values
494  * in excess of @p CH_CFG_ST_TIMEDELTA ticks.
495  * @note The interval returned by this function is only meaningful if
496  * more timers are not added to the list until the returned time.
497  *
498  * @param[out] timep pointer to a variable that will contain the time
499  * interval until the next timer elapses. This pointer
500  * can be @p NULL if the information is not required.
501  * @return The time, in ticks, until next time event.
502  * @retval false if the timers list is empty.
503  * @retval true if the timers list contains at least one timer.
504  *
505  * @iclass
506  */
507 static inline bool chVTGetTimersStateI(systime_t *timep) {
508 
509  chDbgCheckClassI();
510 
511  if (&ch.vtlist == (virtual_timers_list_t *)ch.vtlist.next) {
512  return false;
513  }
514 
515  if (timep != NULL) {
516 #if CH_CFG_ST_TIMEDELTA == 0
517  *timep = ch.vtlist.next->delta;
518 #else
519  *timep = ch.vtlist.lasttime + ch.vtlist.next->delta +
520  CH_CFG_ST_TIMEDELTA - chVTGetSystemTimeX();
521 #endif
522  }
523 
524  return true;
525 }

657 /**
658  * @brief Virtual timers ticker.
659  * @note The system lock is released before entering the callback and
660  * re-acquired immediately after. It is callback's responsibility
661  * to acquire the lock if needed. This is done in order to reduce
662  * interrupts jitter when many timers are in use.
663  *
664  * @iclass
665  */
666 static inline void chVTDoTickI(void) {
667 
668  chDbgCheckClassI();
669 
670 #if CH_CFG_ST_TIMEDELTA == 0
671  ch.vtlist.systime++;
672  if (&ch.vtlist != (virtual_timers_list_t *)ch.vtlist.next) {
673  /* The list is not empty, processing elements on top.*/
674  --ch.vtlist.next->delta;
675  while (ch.vtlist.next->delta == (systime_t)0) {
676  virtual_timer_t *vtp;
677  vtfunc_t fn;
678 
679  vtp = ch.vtlist.next;
680  fn = vtp->func;
681  vtp->func = NULL;
682  vtp->next->prev = (virtual_timer_t *)&ch.vtlist;
683  ch.vtlist.next = vtp->next;
684  chSysUnlockFromISR();
685  fn(vtp->par);
686  chSysLockFromISR();
687  }
688  }
689 #else /* CH_CFG_ST_TIMEDELTA > 0 */
690  virtual_timer_t *vtp;
691  systime_t now, delta;
692 
693  /* First timer to be processed.*/
694  vtp = ch.vtlist.next;
695  now = chVTGetSystemTimeX();
696 
697  /* All timers within the time window are triggered and removed,
698  note that the loop is stopped by the timers header having
699  "ch.vtlist.vt_delta == (systime_t)-1" which is greater than
700  all deltas.*/
701  while (vtp->delta <= (systime_t)(now - ch.vtlist.lasttime)) {
702  vtfunc_t fn;
703 
704  /* The "last time" becomes this timer's expiration time.*/
705  ch.vtlist.lasttime += vtp->delta;
706 
707  vtp->next->prev = (virtual_timer_t *)&ch.vtlist;
708  ch.vtlist.next = vtp->next;
709  fn = vtp->func;
710  vtp->func = NULL;
711 
712  /* if the list becomes empty then the timer is stopped.*/
713  if (ch.vtlist.next == (virtual_timer_t *)&ch.vtlist) {
714  port_timer_stop_alarm();
715  }
716 
717  /* Leaving the system critical zone in order to execute the callback
718  and in order to give a preemption chance to higher priority
719  interrupts.*/
720  chSysUnlockFromISR();
721 
722  /* The callback is invoked outside the kernel critical zone.*/
723  fn(vtp->par);
724 
725  /* Re-entering the critical zone in order to continue the exploration
726  of the list.*/
727  chSysLockFromISR();
728 
729  /* Next element in the list, the current time could have advanced so
730  recalculating the time window.*/
731  vtp = ch.vtlist.next;
732  now = chVTGetSystemTimeX();
733  }
734 
735  /* if the list is empty, nothing else to do.*/
736  if (ch.vtlist.next == (virtual_timer_t *)&ch.vtlist) {
737  return;
738  }
739 
740  /* Recalculating the next alarm time.*/
741  delta = ch.vtlist.lasttime + vtp->delta - now;
742  if (delta < (systime_t)CH_CFG_ST_TIMEDELTA) {
743  delta = (systime_t)CH_CFG_ST_TIMEDELTA;
744  }
745  port_timer_set_alarm(now + delta);
746 
747  chDbgAssert((chVTGetSystemTimeX() - ch.vtlist.lasttime) <=
748  (now + delta - ch.vtlist.lasttime),
749  "exceeding delta");
750 #endif /* CH_CFG_ST_TIMEDELTA > 0 */
751 }
