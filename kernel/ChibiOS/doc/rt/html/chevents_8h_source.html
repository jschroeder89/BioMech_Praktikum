175 /**
176  * @brief Registers an Event Listener on an Event Source.
177  * @details Once a thread has registered as listener on an event source it
178  * will be notified of all events broadcasted there.
179  * @note Multiple Event Listeners can specify the same bits to be ORed to
180  * different threads.
181  *
182  * @param[in] esp pointer to the @p event_source_t structure
183  * @param[out] elp pointer to the @p event_listener_t structure
184  * @param[in] events the mask of events to be ORed to the thread when
185  * the event source is broadcasted
186  *
187  * @api
188  */
189 static inline void chEvtRegisterMask(event_source_t *esp,
190  event_listener_t *elp,
191  eventmask_t events) {
192 
193  chEvtRegisterMaskWithFlags(esp, elp, events, (eventflags_t)-1);
194 }
195 
196 /**
197  * @brief Registers an Event Listener on an Event Source.
198  * @note Multiple Event Listeners can use the same event identifier, the
199  * listener will share the callback function.
200  *
201  * @param[in] esp pointer to the @p event_source_t structure
202  * @param[out] elp pointer to the @p event_listener_t structure
203  * @param[in] event numeric identifier assigned to the Event Listener.
204  * The value must range between zero and the size, in bit,
205  * of the @p eventmask_t type minus one.
206  *
207  * @api
208  */
209 static inline void chEvtRegister(event_source_t *esp,
210  event_listener_t *elp,
211  eventid_t event) {
212 
213  chEvtRegisterMask(esp, elp, EVENT_MASK(event));
214 }
