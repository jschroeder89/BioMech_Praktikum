62 /**
63  * @brief This function implements the idle thread infinite loop.
64  * @details The function puts the processor in the lowest power mode capable
65  * to serve interrupts.<br>
66  * The priority is internally set to the minimum system value so
67  * that this thread is executed only if there are no other ready
68  * threads in the system.
69  *
70  * @param[in] p the thread parameter, unused in this scenario
71  */
72 static void _idle_thread(void *p) {
73 
74  (void)p;
75 
76  while (true) {
77  /*lint -save -e522 [2.2] Apparently no side effects because it contains
78  an asm instruction.*/
79  port_wait_for_interrupt();
80  /*lint -restore*/
81  CH_CFG_IDLE_LOOP_HOOK();
82  }
83 }
84 #endif /* CH_CFG_NO_IDLE_THREAD == FALSE */

90 /**
91  * @brief ChibiOS/RT initialization.
92  * @details After executing this function the current instructions stream
93  * becomes the main thread.
94  * @pre Interrupts must disabled before invoking this function.
95  * @post The main thread is created with priority @p NORMALPRIO and
96  * interrupts are enabled.
97  *
98  * @special
99  */
100 void chSysInit(void) {
101 
102  _scheduler_init();
103  _vt_init();
104  _trace_init();
105 
106 #if CH_DBG_SYSTEM_STATE_CHECK == TRUE
107  ch.dbg.isr_cnt = (cnt_t)0;
108  ch.dbg.lock_cnt = (cnt_t)0;
109 #endif
110 #if CH_CFG_USE_TM == TRUE
111  _tm_init();
112 #endif
113 #if CH_CFG_USE_MEMCORE == TRUE
114  _core_init();
115 #endif
116 #if CH_CFG_USE_HEAP == TRUE
117  _heap_init();
118 #endif
119 #if CH_DBG_STATISTICS == TRUE
120  _stats_init();
121 #endif
122 
123 #if CH_CFG_NO_IDLE_THREAD == FALSE
124  /* Now this instructions flow becomes the main thread.*/
125 #if CH_CFG_USE_REGISTRY == TRUE
126  currp = _thread_init(&ch.mainthread, (const char *)&ch_debug, NORMALPRIO);
127 #else
128  currp = _thread_init(&ch.mainthread, "main", NORMALPRIO);
129 #endif
130 #else
131  /* Now this instructions flow becomes the idle thread.*/
132  currp = _thread_init(&ch.mainthread, "idle", IDLEPRIO);
133 #endif
134 
135 #if CH_DBG_ENABLE_STACK_CHECK == TRUE
136  {
137  /* Setting up the base address of the static main thread stack, the
138  symbol must be provided externally.*/
139  extern stkalign_t __main_thread_stack_base__;
140  currp->wabase = &__main_thread_stack_base__;
141  }
142 #elif CH_CFG_USE_DYNAMIC == TRUE
143  currp->wabase = NULL;
144 #endif
145 
146  /* Setting up the caller as current thread.*/
147  currp->state = CH_STATE_CURRENT;
148 
149  /* Port layer initialization last because it depend on some of the
150  initializations performed before.*/
151  port_init();
152 
153 #if CH_DBG_STATISTICS == TRUE
154  /* Starting measurement for this thread.*/
155  chTMStartMeasurementX(&currp->stats);
156 #endif
157 
158  /* It is alive now.*/
159  chSysEnable();
160 
161 #if CH_CFG_NO_IDLE_THREAD == FALSE
162  {
163  static const thread_descriptor_t idle_descriptor = {
164  "idle",
165  THD_WORKING_AREA_BASE(ch_idle_thread_wa),
166  THD_WORKING_AREA_END(ch_idle_thread_wa),
167  IDLEPRIO,
168  _idle_thread,
169  NULL
170  };
171 
172  /* This thread has the lowest priority in the system, its role is just to
173  serve interrupts in its context while keeping the lowest energy saving
174  mode compatible with the system status.*/
175  (void) chThdCreate(&idle_descriptor);
176  }
177 #endif
178 }

193

202  /* Halt hook code, usually empty.*/
203  CH_CFG_SYSTEM_HALT_HOOK(reason);

206  while (true) {
207  }

210 /**
211  * @brief System integrity check.
212  * @details Performs an integrity check of the important ChibiOS/RT data
213  * structures.
214  * @note The appropriate action in case of failure is to halt the system
215  * before releasing the critical zone.
216  * @note If the system is corrupted then one possible outcome of this
217  * function is an exception caused by @p NULL or corrupted pointers
218  * in list elements. Exception vectors must be monitored as well.
219  * @note This function is not used internally, it is up to the
220  * application to define if and where to perform system
221  * checking.
222  * @note Performing all tests at once can be a slow operation and can
223  * degrade the system response time. It is suggested to execute
224  * one test at time and release the critical zone in between tests.
225  *
226  * @param[in] testmask Each bit in this mask is associated to a test to be
227  * performed.
228  * @return The test result.
229  * @retval false The test succeeded.
230  * @retval true Test failed.
231  *
232  * @iclass
233  */
234 bool chSysIntegrityCheckI(unsigned testmask) {
235  cnt_t n;
236 
237  chDbgCheckClassI();
238 
239  /* Ready List integrity check.*/
240  if ((testmask & CH_INTEGRITY_RLIST) != 0U) {
241  thread_t *tp;
242 
243  /* Scanning the ready list forward.*/
244  n = (cnt_t)0;
245  tp = ch.rlist.queue.next;
246  while (tp != (thread_t *)&ch.rlist.queue) {
247  n++;
248  tp = tp->queue.next;
249  }
250 
251  /* Scanning the ready list backward.*/
252  tp = ch.rlist.queue.prev;
253  while (tp != (thread_t *)&ch.rlist.queue) {
254  n--;
255  tp = tp->queue.prev;
256  }
257 
258  /* The number of elements must match.*/
259  if (n != (cnt_t)0) {
260  return true;
261  }
262  }
263 
264  /* Timers list integrity check.*/
265  if ((testmask & CH_INTEGRITY_VTLIST) != 0U) {
266  virtual_timer_t * vtp;
267 
268  /* Scanning the timers list forward.*/
269  n = (cnt_t)0;
270  vtp = ch.vtlist.next;
271  while (vtp != (virtual_timer_t *)&ch.vtlist) {
272  n++;
273  vtp = vtp->next;
274  }
275 
276  /* Scanning the timers list backward.*/
277  vtp = ch.vtlist.prev;
278  while (vtp != (virtual_timer_t *)&ch.vtlist) {
279  n--;
280  vtp = vtp->prev;
281  }
282 
283  /* The number of elements must match.*/
284  if (n != (cnt_t)0) {
285  return true;
286  }
287  }
288 
289 #if CH_CFG_USE_REGISTRY == TRUE
290  if ((testmask & CH_INTEGRITY_REGISTRY) != 0U) {
291  thread_t *tp;
292 
293  /* Scanning the ready list forward.*/
294  n = (cnt_t)0;
295  tp = ch.rlist.newer;
296  while (tp != (thread_t *)&ch.rlist) {
297  n++;
298  tp = tp->newer;
299  }
300 
301  /* Scanning the ready list backward.*/
302  tp = ch.rlist.older;
303  while (tp != (thread_t *)&ch.rlist) {
304  n--;
305  tp = tp->older;
306  }
307 
308  /* The number of elements must match.*/
309  if (n != (cnt_t)0) {
310  return true;
311  }
312  }
313 #endif /* CH_CFG_USE_REGISTRY == TRUE */
314 
315 #if defined(PORT_INTEGRITY_CHECK)
316  if ((testmask & CH_INTEGRITY_PORT) != 0U) {
317  PORT_INTEGRITY_CHECK();
318  }
319 #endif
320 
321  return false;
322 }

324 /**
325  * @brief Handles time ticks for round robin preemption and timer increments.
326  * @details Decrements the remaining time quantum of the running thread
327  * and preempts it when the quantum is used up. Increments system
328  * time and manages the timers.
329  * @note The frequency of the timer determines the system tick granularity
330  * and, together with the @p CH_CFG_TIME_QUANTUM macro, the round robin
331  * interval.
332  *
333  * @iclass
334  */
335 void chSysTimerHandlerI(void) {
336 
337  chDbgCheckClassI();
338 
339 #if CH_CFG_TIME_QUANTUM > 0
340  /* Running thread has not used up quantum yet? */
341  if (currp->preempt > (tslices_t)0) {
342  /* Decrement remaining quantum.*/
343  currp->preempt--;
344  }
345 #endif
346 #if CH_DBG_THREADS_PROFILING == TRUE
347  currp->time++;
348 #endif
349  chVTDoTickI();
350  CH_CFG_SYSTEM_TICK_HOOK();
351 }

353 /**
354  * @brief Returns the execution status and enters a critical zone.
355  * @details This functions enters into a critical zone and can be called
356  * from any context. Because its flexibility it is less efficient
357  * than @p chSysLock() which is preferable when the calling context
358  * is known.
359  * @post The system is in a critical zone.
360  *
361  * @return The previous system status, the encoding of this
362  * status word is architecture-dependent and opaque.
363  *
364  * @xclass
365  */
366 syssts_t chSysGetStatusAndLockX(void) {
367 
368  syssts_t sts = port_get_irq_status();
369  if (port_irq_enabled(sts)) {
370  if (port_is_isr_context()) {
371  chSysLockFromISR();
372  }
373  else {
374  chSysLock();
375  }
376  }
377  return sts;
378 }

403 /**
404  * @brief Realtime window test.
405  * @details This function verifies if the current realtime counter value
406  * lies within the specified range or not. The test takes care
407  * of the realtime counter wrapping to zero on overflow.
408  * @note When start==end then the function returns always true because the
409  * whole time range is specified.
410  * @note This function is only available if the port layer supports the
411  * option @p PORT_SUPPORTS_RT.
412  *
413  * @param[in] cnt the counter value to be tested
414  * @param[in] start the start of the time window (inclusive)
415  * @param[in] end the end of the time window (non inclusive)
416  * @retval true current time within the specified time window.
417  * @retval false current time not within the specified time window.
418  *
419  * @xclass
420  */
421 bool chSysIsCounterWithinX(rtcnt_t cnt, rtcnt_t start, rtcnt_t end) {
422 
423  return (bool)((cnt - start) < (end - start));
424 }
