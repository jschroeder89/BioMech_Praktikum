87 /**
88  * @brief Data part of a static memory pool initializer.
89  * @details This macro should be used when statically initializing a
90  * memory pool that is part of a bigger structure.
91  *
92  * @param[in] name the name of the memory pool variable
93  * @param[in] size size of the memory pool contained objects
94  * @param[in] provider memory provider function for the memory pool
95  */
96 #define _MEMORYPOOL_DATA(name, size, provider) \
97  {NULL, size, provider}
98 
99 /**
100  * @brief Static memory pool initializer.
101  * @details Statically initialized memory pools require no explicit
102  * initialization using @p chPoolInit().
103  *
104  * @param[in] name the name of the memory pool variable
105  * @param[in] size size of the memory pool contained objects
106  * @param[in] provider memory provider function for the memory pool or @p NULL
107  * if the pool is not allowed to grow automatically
108  */
109 #define MEMORYPOOL_DECL(name, size, provider) \
110  memory_pool_t name = _MEMORYPOOL_DATA(name, size, provider)
111 
112 #if (CH_CFG_USE_SEMAPHORES == TRUE) || defined(__DOXYGEN__)
113 /**

116  * memory pool that is part of a bigger structure.
117  *
118  * @param[in] name the name of the memory pool variable
119  * @param[in] size size of the memory pool contained objects

169 /**
170  * @brief Adds an object to a memory pool.
171  * @pre The memory pool must be already been initialized.
172  * @pre The added object must be of the right size for the specified
173  * memory pool.
174  * @pre The added object must be memory aligned to the size of
175  * @p stkalign_t type.
176  * @note This function is just an alias for @p chPoolFree() and has been
177  * added for clarity.
178  *
179  * @param[in] mp pointer to a @p memory_pool_t structure
180  * @param[in] objp the pointer to the object to be added
181  *
182  * @api
183  */
