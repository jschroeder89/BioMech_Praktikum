597 static inline void queue_prio_insert(thread_t *tp, threads_queue_t *tqp) {
598 
599  thread_t *cp = (thread_t *)tqp;
600  do {
601  cp = cp->queue.next;
602  } while ((cp != (thread_t *)tqp) && (cp->prio >= tp->prio));
603  tp->queue.next = cp;
604  tp->queue.prev = cp->queue.prev;
605  tp->queue.prev->queue.next = tp;
606  cp->queue.prev = tp;
607 }
