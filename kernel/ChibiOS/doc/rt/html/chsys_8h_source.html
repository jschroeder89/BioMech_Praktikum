179  * @brief Seconds to realtime counter.
180  * @details Converts from seconds to realtime counter cycles.

185  * @return The number of cycles.
186  *
187  * @api
188  */

191 /**
192  * @brief Milliseconds to realtime counter.
193  * @details Converts from milliseconds to realtime counter cycles.
194  * @note The result is rounded upward to the next millisecond boundary.
195  * @note The macro assumes that @p freq >= @p 1000.
196  *
197  * @param[in] freq clock frequency, in Hz, of the realtime counter

381 /**
382  * @brief Leaves the kernel lock state.
383  *
384  * @special
385  */
386 static inline void chSysUnlock(void) {
387 
388  _dbg_check_unlock();
389  _stats_stop_measure_crit_thd();
390 
391  /* The following condition can be triggered by the use of i-class functions
392  in a critical section not followed by a chSchResceduleS(), this means
393  that the current thread has a lower priority than the next thread in
394  the ready list.*/
395  chDbgAssert((ch.rlist.queue.next == (thread_t *)&ch.rlist.queue) ||
396  (ch.rlist.current->prio >= ch.rlist.queue.next->prio),
397  "priority order violation");
398 
399  port_unlock();
400 }

404  * @note This API may do nothing on some architectures, it is required
